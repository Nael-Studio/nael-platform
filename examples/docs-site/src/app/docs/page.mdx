export const cliInstall = `bun install --global @nl-framework/cli
nl new platform-demo
cd platform-demo
bun run dev`;

export const repoInstall = `git clone https://github.com/Nael-Studio/nael-platform.git
cd nael-platform
bun install
bun run build`;

export const gitHistory = `git clone https://github.com/Nael-Studio/nael-platform.git --depth=1 --single-branch
cd nael-platform
bun run --cwd examples/basic-http start`;

export const betterAuthSnippet = `import { BetterAuthModule } from '@nl-framework/auth';

@Module({
imports: [
BetterAuthModule.register({
baseUrl: env.BETTER_AUTH_URL,
multiTenant: true,
}),
],
})
export class AuthModule {}`;

export const graphqlSnippet = `GraphqlModule.forRoot({

  playground: true,
});`;

export const microservicesSnippet = `createMicroservicesModule({
  controllers: [BillingController],
  dapr: {
    pubsubName: 'redis-pubsub',
  },
});`;
