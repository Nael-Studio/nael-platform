import { Badge } from '@/components/ui/badge';
import { CodeBlock } from '@/components/shared/simple-code-block';

export const enumDef = `import { Field, ObjectType, registerEnumType, Resolver, Query } from '@nl-framework/graphql';

export enum Status {
  ACTIVE = 'ACTIVE',
  INACTIVE = 'INACTIVE',
  PENDING = 'PENDING',
}

registerEnumType(Status, {
  name: 'Status',
  description: 'User/account status',
});

@ObjectType()
export class Account {
  @Field()
  id!: string;

  @Field(() => Status)
  status!: Status;
}

@Resolver(() => Account)
export class AccountResolver {
  @Query(() => Account)
  account() {
    return { id: 'acct-1', status: Status.ACTIVE };
  }
}`;

export const enumArgs = `import { Resolver, Mutation, Args } from '@nl-framework/graphql';
import { Status } from './account.enum';

@Resolver()
export class AccountMutations {
  @Mutation(() => Boolean)
  updateStatus(@Args('id') id: string, @Args('status', () => Status) status: Status) {
    // persist
    return true;
  }
}`;
