import { Badge } from '@/components/ui/badge';
import { CodeBlock } from '@/components/shared/simple-code-block';

export const guardExample = `import { Injectable } from '@nl-framework/core';
import type {
  MicroserviceCanActivate,
  MicroserviceExecutionContext,
} from '@nl-framework/microservices';

@Injectable()
export class TenantGuard implements MicroserviceCanActivate {
  canActivate(context: MicroserviceExecutionContext): boolean {
    const { tenantId } = context.getPayload<{ tenantId?: string }>() ?? {};
    return Boolean(tenantId);
  }
}`;

export const controllerSample = `import { Controller, UseGuards } from '@nl-framework/core';
import { MessagePattern } from '@nl-framework/microservices';
import { TenantGuard } from './guards/tenant.guard';

@Controller()
@UseGuards(TenantGuard)
export class BillingConsumer {
  @MessagePattern('billing.charge')
  async charge(payload: { tenantId: string; amount: number }) {
    return { ok: true };
  }

  @MessagePattern('billing.refund')
  @UseGuards(TenantGuard)
  async refund(payload: { tenantId: string; amount: number }) {
    return { ok: true };
  }
}`;

export const globalSample = `import { registerMicroserviceGuard } from '@nl-framework/microservices';

// Runs before every handler's own guards.
registerMicroserviceGuard(AuthGuard);`;
